-- ACTUALIZA FECHA DE VENCIMIENTO CON 30 DESPUES A LA FECHA REAL FILTRADO POR PROVEEDOR --
UPDATE    SAACXP
SET              FECHAV = Y.FECHA_REAL + 30
FROM         SAACXP X INNER JOIN
                      SACOMP_01 Y ON X.NumeroD = Y.NumeroD AND X.TipoCxP = '10' AND Y.TipoCom = 'H'
WHERE     (X.CodProv = 'J303575307')
